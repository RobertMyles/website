function dataViz(incomingData){
        var maxPopulation = d3.max(incomingData, function(el) {
            return parseInt(el.population);});
        var yScale = d3.scale.linear().domain([0,maxPopulation]).range([0,500]);
        
        d3.select("svg").attr("style","height: 500px; width: 600px;");
        
        d3.select("svg")
          .selectAll("rect")
          .data(incomingData)
          .enter()
          .append("rect")

          .attr("height", function(d) {return yScale(parseInt(d.population));})
          .attr("x", function(d,i) {return i * 60;})

          .style("fill", "blue")
          .style("stroke", "red")
